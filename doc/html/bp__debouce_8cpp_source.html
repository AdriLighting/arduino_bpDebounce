224 /**
225  * @fn void bpDebounceHandle::loop()
226  * @brief loop
227  *
228  * @see https://github.com/AdriLighting/
229  * @author Adrien Grellard
230  *
231  */
232 void bpDebounceHandle::loop(){
233  /* chien de garde pour la prise d'etat du bouton */
234  if (_pressedTimer->isActivate()) {_pressedTimer->loop_stop();return;}
235 
236  /* timer now */
237  unsigned long sNow = millis();
238 
239  /* le bouton et pressé */
240  if (_bpDebounce->statuGet()) {
241  switch (_statu) {
242 
243  /* initialisation par default */
244  case mBPD_inactive:
245  _lastChange = sNow; // reset du timerDiff
246  _statu = mBPD_pressed; // le bouton recoit un impulsion
247  _shortPress_count = 0; // remise a 0 compteur de click
248  break;
249 
250  /* 1er impulsion */
251  case mBPD_pressed:
252 
253  /* si l"inpulsion depasse (3000ms) */
254  if ( (sNow-_lastChange) > _longPress_delay ) {
255  _statu = mBPD_detect_long; // le bouton recoit un impulsion > a 3000ms
256  _longPress = true;
257  }
258 
259  /* activation de chien de garde a 100ms si la dernierre impulsion < a 3000ms*/
260  if (!_longPress) {
261  if (!_pressedTimer->isActivate()) {
262  _pressedTimer->set_durationMax(100);
263  _pressedTimer->activate();
264  }
265  }
266  break;
267 
268  /* on ne remet pas le compteur de click a 0 */
269  case mBPD_detect_short:
270  _lastChange = sNow; // reset du timerDiff
271  _statu = mBPD_pressed; // le bouton recoit un impulsion
272  break;
273 
274  /* nada */
275  case mBPD_detect_long: break;
276  default: break;
277  }
278  }
279 
280  /* le bouton est relaché */
281  if (!_bpDebounce->statuGet()) {
282  switch (_statu) {
283 
284 
285  case mBPD_pressed:
286  _shortPress_count++; // incrementation du compteur de click
287  _lastChange = sNow; // reset du timerDiff
288  _statu = mBPD_detect_short; // derinierre impulsion inferieur a 3000ms
289  break;
290 
291  /* si pas d'impulison > 3000ms et que la dernierre impulsion > 250ms on valide le nombre d'inpulsion */
292  case mBPD_detect_short:
293  if ( (sNow - _lastChange) > _shortPress_delay ) {
294  _shortPress = _shortPress_count;
295  _statu = mBPD_inactive;
296  }
297  break;
298 
299  /* si la dernierre impulsion > 3000ms on valide la fin de l'inpulsion longue */
300  case mBPD_detect_long:
301  #ifdef DEBUG
302  Serial.printf_P(PSTR("-%20s [%-3d][%-3d][%-3d]\n"), debug_longPressReleased, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input);
303  #endif
304  if (_callback_lp_e != nullptr) _callback_lp_e->func(); // apell de la fonction callback LC
305  reset_long(); // mise a default des parmaetres
306  break;
307  default: break;
308  }
309  }
310 
311  /* si le bouton n'est pas relacher > 3000ms */
312  if (_statu == mBPD_detect_long) {
313  #ifdef DEBUG
314  Serial.printf_P(PSTR("%-20s [%-3d][%-3d][%-3d]\n"), debug_longPressHandle, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input);
315  #endif
316  if (_callback_lp_l != nullptr) _callback_lp_l->func();
317  }
318 
319  /* si pas d'impulison > 3000ms et que la dernierre impulsion > 250ms on lance le callback atrribuer au nombre de click compter*/
320  if (_shortPress == 1) {
321  #ifdef DEBUG
322  Serial.printf_P(PSTR("%-20s [%-3d][%-3d][%-3d][%d]\n"), debug_shortPress, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input, _shortPress_count);
323  #endif
324  if (_callback_sp_1 != nullptr) _callback_sp_1->func(); // apell de la fonction callback 1CLICK
325  reset_short(); // mise a default des parmaetres
326  }
327  if (_shortPress == 2) {
328  #ifdef DEBUG
329  Serial.printf_P(PSTR("%-20s [%-3d][%-3d][%-3d][%d]\n"), debug_shortPress, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input, _shortPress_count);
330  #endif
331  if (_callback_sp_2 != nullptr) _callback_sp_2->func(); // apell de la fonction callback 2CLIC
332  reset_short(); // mise a default des parmaetres
333  }
334  if (_shortPress == 3) {
335  #ifdef DEBUG
336  Serial.printf_P(PSTR("%-20s [%-3d][%-3d][%-3d][%d]\n"), debug_shortPress, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input, _shortPress_count);
337  #endif
338  if (_callback_sp_3 != nullptr) _callback_sp_3->func(); // apell de la fonction callback 3CLIC
339  reset_short(); // mise a default des parmaetres
340  }
341 }
