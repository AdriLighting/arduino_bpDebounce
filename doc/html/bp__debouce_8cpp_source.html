163 void bpDebounceHandle::loop(){
164 
165  if (_pressedTimer->isActivate()) {_pressedTimer->loop_stop();return;}
166 
167  unsigned long sNow = millis();
168  if (_bpDebounce->statuGet()) {
169  switch (_statu) {
170  case mBPD_inactive:
171  _lastChange = sNow;
172  _statu = mBPD_pressed;
173  _shortPress_count = 0;
174  break;
175  case mBPD_pressed:
176  if ( (sNow-_lastChange) > _longPress_delay ) {
177  _statu = mBPD_detect_long;
178  _longPress = true;
179  }
180  if (!_longPress) {
181  if (!_pressedTimer->isActivate()) {
182  _pressedTimer->set_durationMax(100);
183  _pressedTimer->activate();}
184  }
185  break;
186  case mBPD_detect_short:
187  _lastChange = sNow;
188  _statu = mBPD_pressed;
189  break;
190  case mBPD_detect_long:
191 
192  break;
193  default:
194  break;
195  }
196  }
197  if (!_bpDebounce->statuGet()) {
198  switch (_statu) {
199  case mBPD_pressed:
200  _shortPress_count++;
201  _lastChange = sNow;
202  _statu = mBPD_detect_short;
203  break;
204  case mBPD_detect_short:
205  if ( (sNow - _lastChange) > _shortPress_delay ) {
206  _shortPress = _shortPress_count;
207  _statu = mBPD_inactive;
208  }
209  break;
210  case mBPD_detect_long:
211  #ifdef DEBUG
212  Serial.printf_P(PSTR("-%20s [%-3d][%-3d][%-3d]\n"), debug_longPressReleased, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input);
213  #endif
214  if (_callback_lp_e != nullptr) _callback_lp_e->func();
215  reset_long();
216  break;
217  default:
218  break;
219  }
220  }
221  if (_statu == mBPD_detect_long) {
222  #ifdef DEBUG
223  Serial.printf_P(PSTR("%-20s [%-3d][%-3d][%-3d]\n"), debug_longPressHandle, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input);
224  #endif
225  if (_callback_lp_l != nullptr) _callback_lp_l->func();
226  }
227  if (_shortPress == 1) {
228  #ifdef DEBUG
229  Serial.printf_P(PSTR("%-20s [%-3d][%-3d][%-3d][%d]\n"), debug_shortPress, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input, _shortPress_count);
230  #endif
231  if (_callback_sp_1 != nullptr) _callback_sp_1->func();
232  reset_short();
233  }
234  if (_shortPress == 2) {
235  #ifdef DEBUG
236  Serial.printf_P(PSTR("%-20s [%-3d][%-3d][%-3d][%d]\n"), debug_shortPress, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input, _shortPress_count);
237  #endif
238  if (_callback_sp_2 != nullptr) _callback_sp_2->func();
239  reset_short();
240  }
241  if (_shortPress == 3) {
242  #ifdef DEBUG
243  Serial.printf_P(PSTR("%-20s [%-3d][%-3d][%-3d][%d]\n"), debug_shortPress, _bpDebounce->_pin, _bpDebounce->_pullup, _bpDebounce->_input, _shortPress_count);
244  #endif
245  if (_callback_sp_3 != nullptr) _callback_sp_3->func();
246  reset_short();
247  }
248 }
